#include <iostream>
#include "http_request.hpp" // Incluye el archivo con las definiciones y funciones

int main()
{
    // Realizar la solicitud HTTP
    std::string responseData = performHttpRequest("https://skins-vis.galnod.com/");

    // Extraer el valor de diferentes campos del cuerpo JSON de la respuesta
    std::string applicationDescription = extractField(responseData, "Application-Description");
    std::string applicationName = extractField(responseData, "Application-Name");
    std::string statusCode = extractField(responseData, "Runtime-Mode");

    // Imprimir los valores extraídos
    std::cout << "Application-Description: " << applicationDescription << std::endl;
    std::cout << "Application-Name: " << applicationName << std::endl;
    std::cout << "Status Code: " << statusCode << std::endl;

    return 0;
}
